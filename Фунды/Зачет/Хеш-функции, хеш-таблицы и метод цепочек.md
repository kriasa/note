## 13. Rehashing (–ü–µ—Ä–µ—Ö–µ—à–∏—Ä–æ–≤–∞–Ω–∏–µ)

–ö–æ–≥–¥–∞ Œ± –ø—Ä–µ–≤—ã—à–∞–µ—Ç –ø–æ—Ä–æ–≥–æ–≤–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ, –Ω—É–∂–Ω–æ —É–≤–µ–ª–∏—á–∏—Ç—å —Ä–∞–∑–º–µ—Ä —Ç–∞–±–ª–∏—Ü—ã:

```c

        return;

    table->size = new_size;
    table->count = 0;
    
    // –ü–µ—Ä–µ–º–µ—â–∞–µ–º –≤—Å–µ —ç–ª–µ–º–µ–Ω—Ç—ã –≤ –Ω–æ–≤—É—é —Ç–∞–±–ª–∏—Ü—É

        while (current != NULL) {
            HashNode *next = current->next;
            
            // –í—ã—á–∏—Å–ª—è–µ–º –Ω–æ–≤—ã–π –∏–Ω–¥–µ–∫—Å
            size_t new_index = table->hash_func(current->key, new_size);
            
            // –í—Å—Ç–∞–≤–ª—è–µ–º –≤ –Ω–æ–≤—É—é —Ç–∞–±–ª–∏—Ü—É (–≤ –Ω–∞—á–∞–ª–æ —Å–ø–∏—Å–∫–∞)

    // –û—Å–≤–æ–±–æ–∂–¥–∞–µ–º —Å—Ç–∞—Ä—ã–π –º–∞—Å—Å–∏–≤ (–Ω–æ –Ω–µ —É–∑–ª—ã!)

        // –£–¥–≤–∞–∏–≤–∞–µ–º —Ä–∞–∑–º–µ—Ä —Ç–∞–±–ª–∏—Ü—ã

}
```
