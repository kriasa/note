### 2. –°—Ç—Ä–∞—Ç–µ–≥–∏—è —Ñ–∏–∫—Å–∏—Ä–æ–≤–∞–Ω–Ω–æ–≥–æ –ø—Ä–∏—Ä–∞—â–µ–Ω–∏—è (Additive Growth)

// 1. –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è: –°–æ–∑–¥–∞–Ω–∏–µ –ø—É—Å—Ç–æ–≥–æ –º–∞—Å—Å–∏–≤–∞ —Å –Ω–∞—á–∞–ª—å–Ω–æ–π –µ–º–∫–æ—Å—Ç—å—é
// –°–ª–æ–∂–Ω–æ—Å—Ç—å: O(1)
int array_init(DynamicArray *arr) {
    arr->capacity = 8; // –ù–∞—á–∞–ª—å–Ω–∞—è –µ–º–∫–æ—Å—Ç—å, —á—Ç–æ–±—ã –∏–∑–±–µ–∂–∞—Ç—å —Ä–µ–∞–ª–ª–æ–∫–∞—Ü–∏–∏ —Å—Ä–∞–∑—É
    arr->size = 0;     // –ò–∑–Ω–∞—á–∞–ª—å–Ω–æ –º–∞—Å—Å–∏–≤ –ø—É—Å—Ç
    arr->data = (int *)malloc(arr->capacity * sizeof(int)); // –í—ã–¥–µ–ª–µ–Ω–∏–µ –ø–∞–º—è—Ç–∏
    
    if (arr->data == NULL) return 0; // –û—à–∏–±–∫–∞ –≤—ã–¥–µ–ª–µ–Ω–∏—è –ø–∞–º—è—Ç–∏
    return 1;
}

// 2. –î–æ—Å—Ç—É–ø –∫ —ç–ª–µ–º–µ–Ω—Ç—É –ø–æ –∏–Ω–¥–µ–∫—Å—É (Random Access)
// –°–ª–æ–∂–Ω–æ—Å—Ç—å: O(1)
int array_get(DynamicArray *arr, size_t index, int *value) {
    if (index >= arr->size) return 0; // –ü—Ä–æ–≤–µ—Ä–∫–∞ –≥—Ä–∞–Ω–∏—Ü (Production-ready)
    *value = arr->data[index]; // –ü—Ä—è–º–æ–µ –æ–±—Ä–∞—â–µ–Ω–∏–µ –ø–æ –∞–¥—Ä–µ—Å—É
    return 1;
}

// 3. –î–æ–±–∞–≤–ª–µ–Ω–∏–µ —ç–ª–µ–º–µ–Ω—Ç–∞ –≤ –∫–æ–Ω–µ—Ü (push_back)
// –°–ª–æ–∂–Ω–æ—Å—Ç—å: O(1) –∞–º–æ—Ä—Ç–∏–∑–∏—Ä–æ–≤–∞–Ω–Ω–∞—è
int array_push_back(DynamicArray *arr, int value) {
    // –ï—Å–ª–∏ –º–µ—Å—Ç–∞ –Ω–µ—Ç ‚Äî —Ä–µ–∞–ª–ª–æ—Ü–∏—Ä–æ–≤–∞—Ç—å —Å —É–≤–µ–ª–∏—á–µ–Ω–Ω–æ–π –µ–º–∫–æ—Å—Ç—å—é
    if (arr->size == arr->capacity) {
        size_t new_capacity = arr->capacity * 2; // –£–¥–≤–æ–µ–Ω–∏–µ –µ–º–∫–æ—Å—Ç–∏
        int *new_data = (int *)realloc(arr->data, new_capacity * sizeof(int));
        if (new_data == NULL) return 0;
        arr->data = new_data;
        arr->capacity = new_capacity;
    }
    
    arr->data[arr->size] = value; // –ó–∞–ø–∏—Å—å —ç–ª–µ–º–µ–Ω—Ç–∞
    arr->size++; // –ò–Ω–∫—Ä–µ–º–µ–Ω—Ç —Ä–∞–∑–º–µ—Ä–∞
    return 1;
}

// 4. –î–æ–±–∞–≤–ª–µ–Ω–∏–µ –≤ —Å–µ—Ä–µ–¥–∏–Ω—É (insert)
// –°–ª–æ–∂–Ω–æ—Å—Ç—å: O(n) –∏–∑-–∑–∞ –Ω–µ–æ–±—Ö–æ–¥–∏–º–æ—Å—Ç–∏ —Å–¥–≤–∏–≥–∞
int array_insert(DynamicArray *arr, size_t index, int value) {
    if (index > arr->size) return 0; // –ü—Ä–æ–≤–µ—Ä–∫–∞ –≥—Ä–∞–Ω–∏—Ü

    // –ï—Å–ª–∏ –º–∞—Å—Å–∏–≤ –ø–æ–ª–æ–Ω ‚Äî —Ä–∞—Å—à–∏—Ä—è–µ–º
    if (arr->size == arr->capacity) {
        size_t new_capacity = arr->capacity * 2;
        int *new_data = (int *)realloc(arr->data, new_capacity * sizeof(int));
        if (new_data == NULL) return 0;
        arr->data = new_data;
        arr->capacity = new_capacity;
    }

    // –°–¥–≤–∏–≥ —ç–ª–µ–º–µ–Ω—Ç–æ–≤ –≤–ø—Ä–∞–≤–æ, –Ω–∞—á–∏–Ω–∞—è —Å –∫–æ–Ω—Ü–∞
    for (size_t i = arr->size; i > index; i--) {
        arr->data[i] = arr->data[i - 1];
    }

    arr->data[index] = value; // –ó–∞–ø–∏—Å—å –Ω–∞ –ø–æ–∑–∏—Ü–∏—é
    arr->size++;
    return 1;
}

// 5. –£–¥–∞–ª–µ–Ω–∏–µ –∏–∑ –∫–æ–Ω—Ü–∞ (pop_back)
// –°–ª–æ–∂–Ω–æ—Å—Ç—å: O(1)
int array_pop_back(DynamicArray *arr) {
    if (arr->size == 0) return 0; // –ü—Ä–æ–≤–µ—Ä–∫–∞, —á—Ç–æ –º–∞—Å—Å–∏–≤ –Ω–µ –ø—É—Å—Ç
    arr->size--; // –ü—Ä–æ—Å—Ç–æ —É–º–µ–Ω—å—à–∞–µ–º —Å—á–µ—Ç—á–∏–∫
    return 1;
}

// 6. –£–¥–∞–ª–µ–Ω–∏–µ –∏–∑ —Å–µ—Ä–µ–¥–∏–Ω—ã (erase)
// –°–ª–æ–∂–Ω–æ—Å—Ç—å: O(n) –∏–∑-–∑–∞ —Å–¥–≤–∏–≥–∞ —ç–ª–µ–º–µ–Ω—Ç–æ–≤ –≤–ª–µ–≤–æ
int array_erase(DynamicArray *arr, size_t index) {
    if (index >= arr->size) return 0; // –ü—Ä–æ–≤–µ—Ä–∫–∞ –≥—Ä–∞–Ω–∏—Ü

    // –°–¥–≤–∏–≥ –≤—Å–µ—Ö —ç–ª–µ–º–µ–Ω—Ç–æ–≤ –ø–æ—Å–ª–µ –∏–Ω–¥–µ–∫—Å–∞ –≤–ª–µ–≤–æ
    for (size_t i = index; i < arr->size - 1; i++) {
        arr->data[i] = arr->data[i + 1];
    }
    
    arr->size--; // –î–µ–∫—Ä–µ–º–µ–Ω—Ç —Ä–∞–∑–º–µ—Ä–∞
    return 1;
}

// 7. –ü–æ–∏—Å–∫ –ø–æ –∑–Ω–∞—á–µ–Ω–∏—é (find)
// –°–ª–æ–∂–Ω–æ—Å—Ç—å: O(n) (–õ–∏–Ω–µ–π–Ω—ã–π –ø–æ–∏—Å–∫)
int array_find(DynamicArray *arr, int value) {
    for (size_t i = 0; i < arr->size; i++) {
        if (arr->data[i] == value) return (int)i; // –í–æ–∑–≤—Ä–∞—â–∞–µ–º –∏–Ω–¥–µ–∫—Å
    }
    return -1; // –ù–µ –Ω–∞–π–¥–µ–Ω–æ
}

void array_clear(DynamicArray *arr) {
    arr->size = 0; // –ü–∞–º—è—Ç—å –æ—Å—Ç–∞–µ—Ç—Å—è, –Ω–æ –º–∞—Å—Å–∏–≤ –ª–æ–≥–∏—á–µ—Å–∫–∏ –ø—É—Å—Ç
}

// 9. –£–Ω–∏—á—Ç–æ–∂–µ–Ω–∏–µ –º–∞—Å—Å–∏–≤–∞ (destroy)
// –°–ª–æ–∂–Ω–æ—Å—Ç—å: O(1)
void array_destroy(DynamicArray *arr) {
    free(arr->data); // –û—Å–≤–æ–±–æ–∂–¥–µ–Ω–∏–µ –ø–∞–º—è—Ç–∏
    arr->data = NULL;

    arr->capacity = 0;
}
